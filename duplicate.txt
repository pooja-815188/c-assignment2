#include<stdio.h>
int main()
{
    int n,i,j,count=0;
    printf("enter a size\n");
    scanf("%d",&n);
    int arr[n];
    printf("enter a array elemet\n");
    for(i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
    }
    printf("duplicate elemets are\n");
    for(i=0;i<n;i++)
    {
        for(j=i+1;j<n;j++)
        { 
           if(arr[i]==arr[j])
             {
               count++;
               break;
            }
        }
    }
    printf("count=%d\n",count);
}
